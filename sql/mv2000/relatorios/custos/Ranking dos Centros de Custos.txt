 SELECT PERIODO,
 CD_SETOR,
 NM_SETOR,
 CUSTO_TOTAL
 FROM(
 SELECT R.PERIODO,
 R.CD_SETOR,
 R.NM_SETOR,
 SUM(R.CUSTO_TOTAL) CUSTO_TOTAL
 FROM RANKING_CUSTO_SETOR R
 WHERE R.PERIODO=#MESANO#
 AND  r.cd_setor in (55,56,57,58,59,60,61,62,63,64,65,66,67,68,137,71)
 GROUP BY
 R.PERIODO,
 R.CD_SETOR,
 R.NM_SETOR
    UNION
  SELECT TO_CHAR(NULL)PERIODO,
 TO_NUMBER(NULL)CD_SETOR,
 R.PERIODO NM_SETOR,
 TO_NUMBER(NULL) CUSTO_TOTAL
 FROM RANKING_CUSTO_SETOR R
 WHERE R.PERIODO=#MESANO#
 group by R.PERIODO
  UNION
  SELECT TO_CHAR(NULL)PERIODO,
 TO_NUMBER(NULL)CD_SETOR,
 'OUTROS CENTROS' NM_SETOR,
 SUM(R.CUSTO_TOTAL) CUSTO_TOTAL
 FROM RANKING_CUSTO_SETOR R
 WHERE R.PERIODO=#MESANO#
 AND  r.cd_setor not in (55,57,58,59,60,61,62,63,64,65,66,67,68,137,71)
  UNION
  SELECT TO_CHAR(NULL)PERIODO,
 TO_NUMBER(NULL)CD_SETOR,
 'TOTAL DOS CENTROS DE CUSTO' NM_SETOR,
 SUM(R.CUSTO_TOTAL) CUSTO_TOTAL
 FROM RANKING_CUSTO_SETOR R
 WHERE R.PERIODO=#MESANO#
   UNION
  SELECT TO_CHAR(NULL)PERIODO,
 TO_NUMBER(NULL)CD_SETOR,
 R.PERIODO NM_SETOR,
 TO_NUMBER(NULL) CUSTO_TOTAL
 FROM RANKING_CUSTO_SETOR R
 WHERE R.PERIODO=#MESANO#
 group by R.PERIODO

 )
  ORDER BY 4 DESC
